35/// Deserializes an optional string field and parses it into an Uuid, doing this
36/// optionally without failure. If an error occurs, a None is returned. This is
37/// desired for the page_id field in DocumentLookupArguments, which is generated
38/// by an LLM and can be None or a non-Uuid string in some cases. If any errors
39/// should occur, they are emitted in ChatbotTools's from_db_and_arguments.
40fn deserialize_to_optional_uuid_and_errors_to_none<'de, D>(
41    deserializer: D,
42) -> Result<Option<Uuid>, D::Error>
43where
44    D: Deserializer<'de>,
45{
46    let res = String::deserialize(deserializer)
47        .ok()
48        .and_then(|s| Uuid::from_str(&s).ok());
49    Ok(res)
50}

52fn shorten_page_content(content: String) -> String {
53    let page_tokens = estimate_tokens(&content);
54    if page_tokens <= 25000 {
55        return content.to_string();
56    }
57    // a token is ~4 chars and a char is ~4 bytes
58    let max_bytes = ((page_tokens - 1000) * 4 * 4) as usize;
59
60    let shortened = truncate_utf8_at_boundary(&content, max_bytes);
61    shorten_page_content(shortened.to_string())
62}

64/// Look up a document (page) from the course the chatbot is on.
65impl ChatbotTool for DocumentLookupTool {
66    type State = DocumentLookupState;
67    type Arguments = DocumentLookupArguments;
68
69    fn parse_arguments(args_string: String) -> ChatbotResult<Self::Arguments> {
70        serde_json::from_str::<Self::Arguments>(&args_string).map_err(|e| {
71            chatbot_err!(
72                InvalidToolArguments,
73                format!("Couldn't parse tool arguments. Arguments: {args_string}"),
74                e
75            )
76        })
77    }
78
79    async fn from_db_and_arguments(
80        conn: &mut PgConnection,
81        mut arguments: Self::Arguments,
82        user_context: &ChatbotUserContext,
83    ) -> ChatbotResult<Self> {
84        let page_id = if let Some(id) = &arguments.page_id {
85            id.to_owned()
86        } else if let Some(f) = &arguments.filepath {
87            let res = parse_document_filepath(f);
88            match res {
89                Ok(d) => d.page_id,
90                Err(e) => Err(chatbot_err!(
91                    InvalidToolArguments,
92                    "Couldn't parse document file path and no valid page id was provided, unable to look up document.",
93                    e
94                ))?,
95            }
96        } else {
97            return Err(chatbot_err!(
98                InvalidToolArguments,
99                format!(
100                    "Unable to call document_lookup tool. No filepath or page id provided. One of them is needed to find the document."
101                )
102            ));
103        };
104        let course_id = user_context.course_id;
105        let page_content =
106            headless_lms_models::course_page_markdown_content::get_course_page_content_by_page_id(
107                conn, page_id,
108            )
109            .await?;
110
111        let document =
112            // Check if the titles match and the page is part of the same course as
113            // the one the user is on.
114            if page_content.course_id == course_id {
115                arguments.title = page_content.title;
116                if arguments.format == "json" {
117                    let s = shorten_page_content(serde_json::to_string(&page_content.json_content)?);
118                    Some(s)
119                } else {
120                // use markdown content if there is any. else use json as string
121                if let Some(content) = page_content.markdown_content {
122                    let s = shorten_page_content(content);
123                    Some(s)
124                } else {
125                    let base = "Markdown content not found. Page JSON content:\n\n".to_string();
126                    let s = shorten_page_content(serde_json::to_string(&page_content.json_content)?);
127                    Some(base + &s)
128                }
129                }
130
131            } else {
132                None
133            };
134
135        Ok(DocumentLookupTool {
136            state: DocumentLookupState { document },
137            arguments,
138        })
139    }
140
141    fn output(&self) -> String {
142        if let Some(d) = &self.state.document {
143            d.to_string()
144        } else {
145            "Document not found.".to_string()
146        }
147    }
148
149    fn output_description_instructions(&self) -> Option<String> {
150        Some("Do not return the whole document to the user. Use the document as a source of more information for answering the user etc. If you need to cite the content of this document, cite the Azure search result of the document.".to_string())
151    }
152
153    fn get_arguments(&self) -> &Self::Arguments {
154        &self.arguments
155    }
