15#[derive(Serialize, Deserialize, Debug, Clone)]
16pub struct CourseMaterialDocument {
17    pub chunk_id: String,
18    pub chunk: String,
19    pub title: String,
20    pub url: String,
21    pub filepath: String,
22}
23
24impl CourseMaterialDocument {
25    /// Converts the document to citation. Returns also the page_id of the cited document
26    /// so we can get the correct chapter_number later.
27    pub fn to_chatbot_conversation_message_citation(
28        &self,
29        conversation_message_id: Uuid,
30        conversation_id: Uuid,
31        citation_number: i32,
32    ) -> ChatbotResult<(ChatbotConversationMessageCitation, Option<Uuid>)> {
33        // Shorten the content if needed
34        let content = if self.chunk.len() < 255 {
35            self.chunk.clone()
36        } else {
37            truncate_utf8_at_boundary(&self.chunk, 255).to_string()
38        };
39
40        // The title and URL come from Azure Blob Storage metadata, which was URL-encoded
41        // (percent-encoded) because Azure Blob Storage metadata values must be ASCII-only.
42        // We decode them back to their original UTF-8 strings before storing in the database.
43        let decoded_title = url_decode(&self.title)?;
44        let decoded_url = url_decode(&self.url)?;
45
46        // Get the page id
47        let mut page_path = PathBuf::from(&self.filepath);
48        page_path.set_extension("");
49        let page_id_str = page_path.file_name();
50        let page_id =
51            page_id_str.and_then(|id_str| Uuid::parse_str(id_str.to_string_lossy().as_ref()).ok());
52
53        Ok((
54            ChatbotConversationMessageCitation {
55                conversation_message_id,
56                conversation_id,
57                title: decoded_title,
58                content,
59                document_url: decoded_url,
60                citation_number,
61                ..Default::default()
62            },
63            page_id,
64        ))
65    }
66}

68/// Get documents cited by the chatbot from the search index and save them
69/// as chatbot_conversation_message_citations into the database
70pub async fn chatbot_cited_documents_to_citations(
71    conn: &mut PgConnection,
72    test_chatbot: bool,
73    mut document_urls: Vec<Url>,
74    api_key: &str,
75    conversation_message_id: Uuid,
76    conversation_id: Uuid,
77) -> anyhow::Result<Vec<ChatbotConversationMessageCitation>> {
78    let mut documents: Vec<(CourseMaterialDocument, i32)> = vec![];
79    for (idx, url) in document_urls.iter_mut().enumerate() {
80        let document = get_course_material_document(url, api_key).await?;
81        let citation_number = (idx + 1) as i32;
82        documents.push((document, citation_number));
83    }
84    let res = save_documents(
85        conn,
86        test_chatbot,
87        documents,
88        conversation_message_id,
89        conversation_id,
90    )
91    .await?;
92
93    Ok(res)
94}

96/// Get a document from the search index with a LLM-provided get url
97async fn get_course_material_document(
98    endpoint: &mut Url,
99    api_key: &str,
100) -> anyhow::Result<CourseMaterialDocument> {
101    endpoint.set_query(Some(
102        "api-version=2024-07-01&$select=chunk_id,parent_id,chunk,title,url,filepath,course_id",
103    ));
104    let headers = build_llm_headers(api_key)?;
105
106    let response = REQWEST_CLIENT
107        .get(endpoint.clone())
108        .headers(headers)
109        .send()
110        .await?;
111
112    process_course_material_document_response(response).await
113}
114
115#[instrument(skip(response), fields(status = %response.status()))]
116async fn process_course_material_document_response(
117    response: Response,
118) -> anyhow::Result<CourseMaterialDocument> {
119    if !response.status().is_success() {
120        let status = response.status();
121        let error_text = response.text().await?;
122        error!(
123            status = %status,
124            error = %error_text,
125            "Error fetching document from search index."
126        );
127        return Err(anyhow::anyhow!(
128            "Error fetching document from search index: Status: {}. Error: {}",
129            status,
130            error_text
131        ));
132    }
133
134    trace!("Processing successful LLM response");
135    // Parse the response
136    let document: CourseMaterialDocument = response.json().await?;
137
138    Ok(document)
139}

141/// Save a course material document into the database as a citation
142async fn save_documents(
143    conn: &mut PgConnection,
144    test_chatbot: bool,
145    documents_with_citation_numbers: Vec<(CourseMaterialDocument, i32)>,
146    conversation_message_id: Uuid,
147    conversation_id: Uuid,
148) -> anyhow::Result<Vec<ChatbotConversationMessageCitation>> {
149    let (citations, page_ids): (Vec<ChatbotConversationMessageCitation>, Vec<Option<Uuid>>) =
150        documents_with_citation_numbers
151            .iter()
152            .map(|(d, citation_number)| {
153                d.to_chatbot_conversation_message_citation(
154                    conversation_message_id,
155                    conversation_id,
156                    citation_number.to_owned(),
157                )
158            })
159            .collect::<ChatbotResult<Vec<(ChatbotConversationMessageCitation, Option<Uuid>)>>>()?
160            .into_iter()
161            .unzip();
162    if test_chatbot {
163        return save_documents_mock(conn, citations).await;
164    };
165    let res =
166        chatbot_conversation_messages_citations::insert_batch(conn, citations, page_ids).await?;
167
168    Ok(res)
169}

171async fn save_documents_mock(
