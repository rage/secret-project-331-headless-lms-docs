29/**
30Generates a certificate as a png.
31
32## Arguments
33
34- debug: If true, the certificate will have a red dot at the anchoring points, and the URL will be replaced with a placeholder.
35  Should be false when rendering certificates for the students. However, when positioning the texts, this can be used to see why the texts were positioned where they were.
36*/
37#[allow(clippy::too_many_arguments)]
38pub async fn generate_certificate(
39    conn: &mut PgConnection,
40    file_store: &dyn FileStore,
41    certificate: &GeneratedCertificate,
42    debug: bool,
43    icu4x_blob: Icu4xBlob,
44) -> UtilResult<Vec<u8>> {
45    let config = headless_lms_models::certificate_configurations::get_by_id(
46        &mut *conn,
47        certificate.certificate_configuration_id,
48    )
49    .await
50    .map_err(|original_error| {
51        UtilError::new(
52            UtilErrorType::Other,
53            "Certificate configuration not found".to_string(),
54            Some(original_error.into()),
55        )
56    })?;
57    let background_svg = file_store
58        .fetch_file_content_or_use_filesystem_cache(Path::new(&config.background_svg_path))
59        .await?;
60    let overlay_svg = OptionFuture::from(
61        config
62            .overlay_svg_path
63            .as_ref()
64            .map(|path| file_store.fetch_file_content_or_use_filesystem_cache(Path::new(path))),
65    )
66    .await
67    .transpose()?;
68
69    let grade = if config.render_certificate_grade {
70        let requirements = headless_lms_models::certificate_configuration_to_requirements::get_all_requirements_for_certificate_configuration(
71        conn,
72        certificate.certificate_configuration_id,
73    )
74    .await
75    .map_err(|original_error| {
76        UtilError::new(
77            UtilErrorType::Other,
78            "No certificate conf requirements".to_string(),
79            Some(original_error.into()),
80        )
81    })?;
82
83        if let Some(course_module_id) = requirements.course_module_ids.first() {
84            let completions = headless_lms_models::course_module_completions::get_all_by_user_id_and_course_module_id(
85        conn,
86        certificate.user_id,
87        *course_module_id,
88    )
89    .await
90    .map_err(|original_error| {
91        UtilError::new(
92            UtilErrorType::Other,
93            "No completion found for user".to_string(),
94            Some(original_error.into()),
95        )
96    })?;
97
98            let grade_option =
99                headless_lms_models::course_module_completions::select_best_completion(completions);
100
101            rust_i18n::set_locale(&config.certificate_locale);
102            let grade_label = t!("grade");
103
104            grade_option.map(|grade| {
105                if let Some(numeral_grade) = grade.grade {
106                    format!("{} {}", grade_label, numeral_grade)
107                } else {
108                    let passed_text = if grade.passed {
109                        t!("passed")
110                    } else {
111                        t!("failed")
112                    };
113                    format!("{} {}", grade_label, passed_text)
114                }
115            })
116        } else {
117            None
118        }
119    } else {
120        None
121    };
122
123    let fontdb = font_loader::get_font_database_with_fonts(&mut *conn, file_store).await?;
124    let url = if debug {
125        "https://courses.mooc.fi/certificates/validate/debug".to_string()
126    } else {
127        format!(
128            // TODO: use base url here
129            "https://courses.mooc.fi/certificates/validate/{}",
130            certificate.verification_id
131        )
132    };
133    let date = if debug {
134        // TODO: this fixes the date for system tests, not a great solution...
135        NaiveDate::from_ymd_opt(2023, 1, 1).ok_or_else(|| {
136            UtilError::new(
137                UtilErrorType::Other,
138                "Invalid date 2023-01-01 - this should never happen".to_string(),
139                None,
140            )
141        })?
142    } else {
143        certificate.created_at.date_naive()
144    };
145    let mut texts_to_render = vec![
146        TextToRender {
147            text: certificate.name_on_certificate.to_string(),
148            y_pos: config.certificate_owner_name_y_pos,
149            x_pos: config.certificate_owner_name_x_pos,
150            font_size: config.certificate_owner_name_font_size,
151            text_anchor: config.certificate_owner_name_text_anchor,
152            text_color: config.certificate_owner_name_text_color,
153            ..Default::default()
154        },
155        TextToRender {
156            text: url,
157            y_pos: config.certificate_validate_url_y_pos,
158            x_pos: config.certificate_validate_url_x_pos,
159            font_size: config.certificate_validate_url_font_size,
160            text_anchor: config.certificate_validate_url_text_anchor,
161            text_color: config.certificate_validate_url_text_color,
162            ..Default::default()
163        },
164        TextToRender {
165            text: get_date_as_localized_string(&config.certificate_locale, date, icu4x_blob)?,
166            y_pos: config.certificate_date_y_pos,
167            x_pos: config.certificate_date_x_pos,
168            font_size: config.certificate_date_font_size,
169            text_anchor: config.certificate_date_text_anchor,
170            text_color: config.certificate_date_text_color,
171            ..Default::default()
172        },
173    ];
174    if let Some(grade_text) = grade {
175        texts_to_render.push(TextToRender {
176            text: grade_text,
177            x_pos: config.certificate_grade_x_pos.clone().unwrap_or_default(),
178            y_pos: config.certificate_grade_y_pos.clone().unwrap_or_default(),
179            font_size: config
180                .certificate_grade_font_size
181                .clone()
182                .unwrap_or_default(),
183            text_anchor: config
184                .certificate_grade_text_anchor
185                .unwrap_or(CertificateTextAnchor::Middle),
186            text_color: config
187                .certificate_grade_text_color
188                .clone()
189                .unwrap_or_default(),
190            ..Default::default()
191        });
192    }
193    let paper_size = config.paper_size;
194
195    let res = generate_certificate_impl(
196        &background_svg,
197        overlay_svg.as_deref(),
198        &texts_to_render,
199        &paper_size,
200        debug,
201        Arc::new(fontdb),
202    )?;
203    Ok(res)
204}

206fn generate_certificate_impl(
207    background_svg: &[u8],
208    overlay_svg: Option<&[u8]>,
209    texts: &[TextToRender],
210    paper_size: &PaperSize,
211    debug_show_anchoring_points: bool,
212    fontdb: Arc<fontdb::Database>,
213) -> UtilResult<Vec<u8>> {
214    let start_setup = Instant::now();
215    let opt = usvg::Options {
216        // Last-resort fallback; each text element's family is resolved per-string in generate_text_svg.
217        font_family: "Inter Variable".to_string(),
218        dpi: 600.0,
219        image_rendering: usvg::ImageRendering::OptimizeQuality,
220        shape_rendering: usvg::ShapeRendering::GeometricPrecision,
221        text_rendering: usvg::TextRendering::OptimizeLegibility,
222        fontdb: fontdb.clone(),
223        ..Default::default()
224    };
225
226    let mut pixmap = resvg::tiny_skia::Pixmap::new(paper_size.width_px(), paper_size.height_px())
227        .ok_or_else(|| {
228        UtilError::new(
229            UtilErrorType::Other,
230            "Could not create a pixmap".to_string(),
231            None,
232        )
233    })?;
234    info!("Setup time {:?}", start_setup.elapsed());
235    let parse_background_svg_start = Instant::now();
236    let tree = usvg::Tree::from_data(background_svg, &opt).map_err(|original_error| {
237        UtilError::new(
238            UtilErrorType::Other,
239            "Could not parse background svg".to_string(),
240            Some(original_error.into()),
241        )
242    })?;
243
244    info!(
245        "Parse background svg time {:?}",
246        parse_background_svg_start.elapsed()
247    );
248
249    let start_render_background = Instant::now();
250    // Scaling the background to the paper size, if the aspect ratio is wrong (for example if it does not follow the aspect ratio of A4 paper size), the background will get stretched.
251    // If that's the case, the you should fix the background svg.
252    let background_size = tree.size().to_int_size();
253    let x_scale = paper_size.width_px() as f32 / background_size.width() as f32;
254    let y_scale = paper_size.height_px() as f32 / background_size.height() as f32;
255    info!(
256        "Background size {:?}, paper size: {:?}, x_scale: {}, y_scale: {}",
257        background_size, paper_size, x_scale, y_scale
258    );
259    resvg::render(
260        &tree,
261        resvg::tiny_skia::Transform::from_scale(x_scale, y_scale),
262        &mut pixmap.as_mut(),
263    );
264
265    info!(
266        "Render background time {:?}",
267        start_render_background.elapsed()
268    );
269
270    let text_svg_data = generate_text_svg(
271        texts,
272        debug_show_anchoring_points,
273        paper_size,
274        fontdb.as_ref(),
275    )?;
276    info!("{}", String::from_utf8_lossy(&text_svg_data));
277    let parse_text_svg_start = Instant::now();
278    let text_tree = usvg::Tree::from_data(&text_svg_data, &opt).map_err(|original_error| {
279        UtilError::new(
280            UtilErrorType::Other,
281            "Could not parse text svg".to_string(),
282            Some(original_error.into()),
283        )
284    })?;
285
286    info!("Parse text svg time {:?}", parse_text_svg_start.elapsed());
287
288    let render_text_start = Instant::now();
289    resvg::render(
290        &text_tree,
291        tiny_skia::Transform::default(),
292        &mut pixmap.as_mut(),
293    );
294    info!("Render text time {:?}", render_text_start.elapsed());
295
296    if let Some(overlay_svg) = overlay_svg {
297        let start_render_overlay = Instant::now();
298        let overlay_tree = usvg::Tree::from_data(overlay_svg, &opt).map_err(|original_error| {
299            UtilError::new(
300                UtilErrorType::Other,
301                "Could not parse overlay svg".to_string(),
302                Some(original_error.into()),
303            )
304        })?;
305        resvg::render(
306            &overlay_tree,
307            tiny_skia::Transform::default(),
308            &mut pixmap.as_mut(),
309        );
310
311        info!("Overlay time {:?}", start_render_overlay.elapsed());
312    }
313
314    let save_png_start = Instant::now();
315    let png = pixmap.encode_png().map_err(|original_error| {
316        UtilError::new(
317            UtilErrorType::Other,
318            "Could convert pixmap to png".to_string(),
319            Some(original_error.into()),
320        )
321    })?;
322    info!("Save png time {:?}", save_png_start.elapsed());
323    Ok(png)
324}

352/// Picks the loaded font family that can render every non-whitespace character in `text`,
353/// preferring `preferred` and then the other loaded fonts in load order.
354///
355/// Each certificate text element is rendered in a single font: usvg's per-glyph fallback is
356/// unreliable (usvg 0.47 aborts all fallback as soon as one candidate font's reshape changes the
357/// glyph count, and it replaces per-glyph rather than per-cluster), so we resolve the whole string
358/// to one covering font here instead of relying on it. Falls back to `preferred` when nothing
359/// covers the string (best effort — a name mixing scripts no single font covers will have .notdef
360/// glyphs).
361fn resolve_font_family(text: &str, preferred: &str, fontdb: &fontdb::Database) -> String {
362    let mut faces: Vec<&fontdb::FaceInfo> = fontdb.faces().collect();
363    // Stable sort so `preferred` is tried first while the rest keep their load order.
364    faces.sort_by_key(|face| !face_has_family(face, preferred));
365    for face in faces {
366        if face_covers(fontdb, face.id, text) {
367            return face_family_name(face);
368        }
369    }
370    warn!(
371        "No loaded font covers the text {:?}; falling back to '{}', uncovered glyphs will render as boxes",
372        text, preferred
373    );
374    preferred.to_string()
375}
376
377fn face_has_family(face: &fontdb::FaceInfo, family: &str) -> bool {
378    face.families.iter().any(|(name, _)| name == family)
379}
380
381fn face_family_name(face: &fontdb::FaceInfo) -> String {
382    face.families
383        .iter()
384        .find(|(_, lang)| *lang == fontdb::Language::English_UnitedStates)
385        .or_else(|| face.families.first())
386        .map(|(name, _)| name.clone())
387        .unwrap_or_default()
388}
389
390/// Whether the given face has a glyph for every non-whitespace character in `text`.
391fn face_covers(fontdb: &fontdb::Database, id: fontdb::ID, text: &str) -> bool {
392    fontdb
393        .with_face_data(id, |data, index| {
394            match ttf_parser::Face::parse(data, index) {
395                Ok(face) => text
396                    .chars()
397                    .all(|c| c.is_whitespace() || face.glyph_index(c).is_some()),
398                Err(_) => false,
399            }
400        })
401        .unwrap_or(false)
402}

404fn generate_text_svg(
405    texts: &[TextToRender],
406    debug_show_anchoring_points: bool,
407    paper_size: &PaperSize,
408    fontdb: &fontdb::Database,
409) -> UtilResult<Vec<u8>> {
410    let mut writer = Writer::new(Cursor::new(Vec::new()));
411    writer
412        .create_element("svg")
413        .with_attribute(("width", format!("{}px", paper_size.width_px()).as_str()))
414        .with_attribute(("height", format!("{}px", paper_size.height_px()).as_str()))
415        .with_attribute(("xmlns", "http://www.w3.org/2000/svg"))
416        .write_inner_content(|writer| {
417            for text in texts {
418                // Resolve to a single font that covers the whole string (see resolve_font_family).
419                let font_family = resolve_font_family(&text.text, &text.font_family, fontdb);
420                writer
421                    .create_element("text")
422                    .with_attribute(("x", text.x_pos.as_str()))
423                    .with_attribute(("y", text.y_pos.as_str()))
424                    .with_attribute((
425                        "style",
426                        format!(
427                            "font-size: {}; font-family: {}; fill: {}; font-weight: 400;",
428                            text.font_size, font_family, text.text_color
429                        )
430                        .as_str(),
431                    ))
432                    .with_attribute(("text-anchor", text.text_anchor.to_string().as_str()))
433                    .write_text_content(BytesText::from_escaped(&text.text))
434                    .map_err(|_original_error| {
435                        // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
436                        io::Error::other("Could not write text to svg".to_string())
437                    })?;
438
439                if debug_show_anchoring_points {
440                    writer
441                        .create_element("circle")
442                        .with_attribute(("cx", text.x_pos.as_str()))
443                        .with_attribute(("cy", text.y_pos.as_str()))
444                        .with_attribute(("r", "5"))
445                        .with_attribute(("fill", "#da2e2e"))
446                        .write_empty()
447                        .map_err(|_original_error| {
448                            // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
449                            io::Error::other("Could not write debug point to svg".to_string())
450                        })?;
451                    writer
452                        .create_element("circle")
453                        .with_attribute(("cx", text.x_pos.as_str()))
454                        .with_attribute(("cy", text.y_pos.as_str()))
455                        .with_attribute(("r", "3"))
456                        .with_attribute(("fill", "#ff9b9b"))
457                        .write_empty()
458                        .map_err(|_original_error| {
459                            // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
460                            io::Error::other("Could not write debug point to svg".to_string())
461                        })?;
462                }
463            }
464            Ok(())
465        })
466        .map_err(|original_error: std::io::Error| {
467            UtilError::new(
468                UtilErrorType::Other,
469                "Could not write text svg".to_string(),
470                Some(original_error.into()),
471            )
472        })?;
473
474    Ok(writer.into_inner().into_inner())
475}

604    /// Documents the accepted limitation of per-text resolution: a single string mixing scripts
605    /// that no one loaded font covers can't be fully rendered, so it best-effort falls back to the
606    /// preferred font (some glyphs will be .notdef). Rendering it properly would need per-run
607    /// multi-font layout, which usvg does not support (it ignores per-tspan fonts and its own
608    /// per-glyph fallback aborts on the first glyph-count mismatch after a fallback reshape).
609    #[test]
