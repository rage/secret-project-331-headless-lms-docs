38/// Slim module descriptor so the frontend can label per-module completions and show "X of Y modules"
39/// without a separate course-structure fetch. Default (base) module has `name = None`.
40#[derive(Debug, Serialize, Deserialize, PartialEq, Clone, ToSchema)]
41pub struct CourseModuleInfo {
42    pub id: Uuid,
43    pub name: Option<String>,
44    pub order_number: i32,
45    /// Earliest exercise submission by this user in this module. No module "start" is stored, so the
46    /// frontend uses this to infer when an additional module was first worked on. `None` if untouched.
47    pub first_submission_at: Option<DateTime<Utc>>,
48    /// Number of non-deleted exercises in this module. Submission density is divided by this so courses
49    /// of very different size stay comparable. `0` if the module has no chapter-bound exercises.
50    pub exercise_count: i32,
51    /// This user's exercise submissions in this module bucketed by UTC day, ascending. Empty if none.
52    pub daily_submissions: Vec<DailySubmissionCount>,
53}

103/**
104Inserts enrollment if it doesn't exist yet; on conflict updates deleted_at to NULL (upsert).
105
106Handles duplicate submissions (e.g. multiple tabs or parallel requests) by conflicting on (user_id, course_id, course_instance_id).
107*/
108pub async fn insert_enrollment_if_it_doesnt_exist(
109    conn: &mut PgConnection,
110    enrollment: NewCourseInstanceEnrollment,
111) -> ModelResult<CourseInstanceEnrollment> {
112    let enrollment = sqlx::query_as!(
113        CourseInstanceEnrollment,
114        "
115INSERT INTO course_instance_enrollments (user_id, course_id, course_instance_id)
116VALUES ($1, $2, $3)
117ON CONFLICT (user_id, course_id, course_instance_id)
118DO UPDATE SET deleted_at = NULL
119RETURNING *;
120",
121        enrollment.user_id,
122        enrollment.course_id,
123        enrollment.course_instance_id,
124    )
125    .fetch_one(conn)
126    .await?;
127    Ok(enrollment)
128}

130pub async fn insert_enrollment_and_set_as_current(
131    conn: &mut PgConnection,
132    new_enrollment: NewCourseInstanceEnrollment,
133) -> ModelResult<CourseInstanceEnrollment> {
134    let mut tx = conn.begin().await?;
135
136    let enrollment = insert_enrollment_if_it_doesnt_exist(&mut tx, new_enrollment).await?;
137    crate::user_course_settings::upsert_user_course_settings_for_enrollment(&mut tx, &enrollment)
138        .await?;
139    tx.commit().await?;
140
141    Ok(enrollment)
142}

207pub async fn get_course_instance_enrollments_info_for_user(
208    conn: &mut PgConnection,
209    user_id: Uuid,
210) -> ModelResult<CourseInstanceEnrollmentsInfo> {
211    let course_instance_enrollments = get_by_user_id(conn, user_id).await?;
212
213    let course_instance_ids: Vec<Uuid> = course_instance_enrollments
214        .iter()
215        .map(|e| e.course_instance_id)
216        .collect();
217
218    let course_instances = crate::course_instances::get_by_ids(conn, &course_instance_ids).await?;
219
220    let course_ids: Vec<Uuid> = course_instances.iter().map(|e| e.course_id).collect();
221
222    let courses = crate::courses::get_by_ids(conn, &course_ids).await?;
223
224    let course_module_completions =
225        crate::course_module_completions::get_all_by_user_id(conn, user_id).await?;
226
227    // Returns all user course settings because there is always an enrollment for a current course instance (enforced by a database constraint), and all of those are in the course_ids list
228    let user_course_settings =
229        crate::user_course_settings::get_all_by_user_and_multiple_current_courses(
230            conn,
231            &course_ids,
232            user_id,
233        )
234        .await?;
235
236    Ok(CourseInstanceEnrollmentsInfo {
237        course_instance_enrollments,
238        course_instances,
239        courses,
240        user_course_settings,
241        course_module_completions,
242    })

247    first_enrolled_at: Option<DateTime<Utc>>,

250/// Returns one entry per course the user is enrolled in, with aggregated data.
251pub async fn get_course_enrollments_info_for_user(
252    conn: &mut PgConnection,
253    user_id: Uuid,
254) -> ModelResult<CourseEnrollmentsInfo> {
255    let rows = sqlx::query_as!(
256        CourseEnrollmentRow,
257        "
258SELECT course_id, MIN(created_at) AS first_enrolled_at
259FROM course_instance_enrollments
260WHERE user_id = $1 AND deleted_at IS NULL
261GROUP BY course_id
262ORDER BY first_enrolled_at
263        ",
264        user_id
265    )
266    .fetch_all(&mut *conn)
267    .await?;
268
269    let course_ids: Vec<Uuid> = rows.iter().map(|r| r.course_id).collect();
270
271    // Earliest submission per module for this user; module ids are globally unique, so grouping by
272    // module alone is enough. Used to infer when an additional (non-base) module was first worked on.
273    // Restricts to non-deleted exercises/chapters like the exercise-count and daily-submission queries
274    // below, so all three agree on which content is in scope (soft-deleted content stays hidden).
275    let module_first_submission_rows = sqlx::query!(
276        r#"
277SELECT c.course_module_id AS "course_module_id!",
278       MIN(ess.created_at) AS "first_submission_at!"
279FROM exercise_slide_submissions ess
280JOIN exercises e ON e.id = ess.exercise_id AND e.deleted_at IS NULL
281JOIN chapters c ON c.id = e.chapter_id AND c.deleted_at IS NULL
282WHERE ess.user_id = $1 AND ess.course_id = ANY($2) AND ess.deleted_at IS NULL
283GROUP BY c.course_module_id
284        "#,
285        user_id,
286        &course_ids
287    )
288    .fetch_all(&mut *conn)
289    .await?;
290    let first_submission_by_module: HashMap<Uuid, DateTime<Utc>> = module_first_submission_rows
291        .into_iter()
292        .map(|r| (r.course_module_id, r.first_submission_at))
293        .collect();
294
295    // Exercise count per module, to normalize submission density (submissions per exercise) so courses
296    // of very different size are comparable. Exercises map to a module via their chapter; exercises with
297    // no chapter (e.g. exams) are not counted.
298    let module_exercise_count_rows = sqlx::query!(
299        r#"
300SELECT c.course_module_id AS "course_module_id!",
301       COUNT(*) AS "count!"
302FROM exercises e
303JOIN chapters c ON c.id = e.chapter_id AND c.deleted_at IS NULL
304WHERE e.course_id = ANY($1) AND e.deleted_at IS NULL
305GROUP BY c.course_module_id
306        "#,
307        &course_ids
308    )
309    .fetch_all(&mut *conn)
310    .await?;
311    let exercise_count_by_module: HashMap<Uuid, i64> = module_exercise_count_rows
312        .into_iter()
313        .map(|r| (r.course_module_id, r.count))
314        .collect();
315
316    // This user's submissions per module bucketed by UTC day, for the activity-density violins. Restricts
317    // to the same non-deleted exercises/chapters as the exercise-count query above so the density
318    // numerator and denominator agree; DATE_TRUNC is anchored to UTC (not the session timezone) so the
319    // buckets line up with the frontend's fixed 24h grid.
320    let module_daily_submission_rows = sqlx::query!(
321        r#"
322SELECT c.course_module_id AS "course_module_id!",
323       DATE_TRUNC('day', ess.created_at AT TIME ZONE 'UTC') AT TIME ZONE 'UTC' AS "day!",
324       COUNT(*) AS "count!"
325FROM exercise_slide_submissions ess
326JOIN exercises e ON e.id = ess.exercise_id AND e.deleted_at IS NULL
327JOIN chapters c ON c.id = e.chapter_id AND c.deleted_at IS NULL
328WHERE ess.user_id = $1 AND ess.course_id = ANY($2) AND ess.deleted_at IS NULL
329GROUP BY c.course_module_id, DATE_TRUNC('day', ess.created_at AT TIME ZONE 'UTC')
330ORDER BY c.course_module_id, "day!"
331        "#,
332        user_id,
333        &course_ids
334    )
335    .fetch_all(&mut *conn)
336    .await?;
337    let mut daily_submissions_by_module: HashMap<Uuid, Vec<DailySubmissionCount>> = HashMap::new();
338    for r in module_daily_submission_rows {
339        daily_submissions_by_module
340            .entry(r.course_module_id)
341            .or_default()
342            .push(DailySubmissionCount {
343                day: r.day,
344                count: r.count as i32,
345            });
346    }
347
348    let course_instance_enrollments = get_by_user_id(&mut *conn, user_id).await?;
349    let all_course_module_completions =
350        crate::course_module_completions::get_all_by_user_id(conn, user_id).await?;
351    let user_course_settings =
352        crate::user_course_settings::get_all_by_user_id(conn, user_id).await?;
353    let courses = crate::courses::get_by_ids(conn, &course_ids).await?;
354    let course_instance_ids: Vec<Uuid> = course_instance_enrollments
355        .iter()
356        .map(|e| e.course_instance_id)
357        .collect();
358    let all_course_instances =
359        crate::course_instances::get_by_ids(conn, &course_instance_ids).await?;
360    let all_course_modules = crate::course_modules::get_by_course_ids(conn, &course_ids).await?;
361
362    let mut course_enrollments = Vec::with_capacity(rows.len());
363    for row in rows {
364        let course = courses
365            .iter()
366            .find(|c| c.id == row.course_id)
367            .cloned()
368            .ok_or_else(|| {
369                crate::ModelError::new(
370                    crate::error::ModelErrorType::NotFound,
371                    "Course not found for enrollment".to_string(),
372                    None,
373                )
374            })?;
375        let course_instances: Vec<_> = all_course_instances
376            .iter()
377            .filter(|ci| ci.course_id == row.course_id)
378            .cloned()
379            .collect();
380        let user_course_settings_for_course = user_course_settings
381            .iter()
382            .find(|ucs| ucs.course_language_group_id == course.course_language_group_id)
383            .cloned();
384        let course_modules = all_course_modules
385            .iter()
386            .filter(|m| m.course_id == row.course_id)
387            .map(|m| CourseModuleInfo {
388                id: m.id,
389                name: m.name.clone(),
390                order_number: m.order_number,
391                first_submission_at: first_submission_by_module.get(&m.id).copied(),
392                exercise_count: exercise_count_by_module.get(&m.id).copied().unwrap_or(0) as i32,
393                daily_submissions: daily_submissions_by_module
394                    .get(&m.id)
395                    .cloned()
396                    .unwrap_or_default(),
397            })
398            .collect();
399        let course_module_completions = all_course_module_completions
400            .iter()
401            .filter(|cmc| cmc.course_id == row.course_id)
402            .cloned()
403            .collect();
404        let course_module_completions_needing_review = all_course_module_completions
405            .iter()
406            .filter(|cmc| cmc.course_id == row.course_id && cmc.needs_to_be_reviewed)
407            .count() as i32;
408        let is_current = user_course_settings_for_course
409            .as_ref()
410            .map(|ucs| ucs.current_course_id == row.course_id)
411            .unwrap_or(false);
412
413        let first_enrolled_at = row.first_enrolled_at.ok_or_else(|| {
414            crate::ModelError::new(
415                crate::error::ModelErrorType::Generic,
416                "first_enrolled_at missing for grouped enrollment row".to_string(),
417                None,
418            )
419        })?;
420
421        course_enrollments.push(CourseEnrollmentInfo {
422            course_id: row.course_id,
423            course,
424            course_instances,
425            user_course_settings: user_course_settings_for_course,
426            course_modules,
427            course_module_completions,
428            course_module_completions_needing_review,
429            first_enrolled_at,
430            is_current,
431        });
432    }
433
434    Ok(CourseEnrollmentsInfo { course_enrollments })
435}
