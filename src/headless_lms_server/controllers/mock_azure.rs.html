114// GET /api/v0/mock_azure/test/v1/responses
115// POST /api/v0/mock_azure/test/v1/responses
116async fn mock_azure_chat_responses(
117    app_conf: web::Data<ApplicationConfiguration>,
118    payload: web::Json<AzureCompletionRequest>,
119) -> ControllerResult<String> {
120    assert!(app_conf.test_chatbot && app_conf.test_mode);
121    let message_suggestion_user_prompt = USER_PROMPT;
122
123    let message_kind = &payload
124        .base
125        .input
126        .last()
127        .ok_or(ControllerError::new(
128            ControllerErrorType::BadRequest,
129            "No messages in request, there should be at least one.",
130            None,
131        ))?
132        .message_type;
133    let message = match message_kind {
134        InputItem::Message {
135            role: _role,
136            content,
137        } => Ok(content.to_owned()),
138        _ => Err(ControllerError::new(
139            ControllerErrorType::BadRequest,
140            "The request had a tool call or tool response. This shouldn't happen when using message suggestion LLM.",
141            None,
142        )),
143    }?.get_content_text();
144
145    let suggest_prompt_match = message
146        .matches(message_suggestion_user_prompt)
147        .collect::<Vec<&str>>();
148    let cms_suggest_match = message.contains(USER_PROMPT_PREFIX);
149    let description_suggestion_match = message.contains(DESCRIPTION_USER_PROMPT);
150    let res = if !suggest_prompt_match.is_empty() {
151        SUGGESTION.to_string()
152    } else if cms_suggest_match {
153        CMS_SUGGESTION.to_string()
154    } else if description_suggestion_match {
155        DESCRIPTION_SUGGESTION.to_string()
156    } else {
157        get_response(app_conf.base_url.clone())?
158    };
159    let token = skip_authorize();
160    token.authorized_ok(res)
161}
